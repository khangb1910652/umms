
CREATE TABLE DICHVUSO 
( MaDV INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY ,  
  TenDV nvarchar2 (255) NOT NULL
);

CREATE TABLE PHANHECHUCNANG
( MaPHCN INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TenPHCN nvarchar2 (255) NOT NULL
);

CREATE TABLE CHUCNANG
( MaChucNang INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TenChucNang nvarchar2 (255) NOT NULL,
  MaPHCN INTEGER,
  MaDV INTEGER,
  FOREIGN KEY (MaPHCN) REFERENCES PHANHECHUCNANG(MaPHCN)ON DELETE CASCADE,
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV)ON DELETE CASCADE
);

CREATE TABLE PHIENBANHDSD
( MaPhienBan INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TieuDePhienBan nvarchar2 (1024) NOT NULL
);

CREATE TABLE LOAIHDSD
( MaLoai INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TenLoai nvarchar2 (255) NOT NULL
);


CREATE TABLE HUONGDANSUDUNG
( MaHDSD INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TieuDeHDSD nvarchar2 (1024) NOT NULL,
  MaLoai INTEGER,
  MaChucNang INTEGER,
  MaDV INTEGER,
  FOREIGN KEY (MaLoai) REFERENCES LOAIHDSD(MaLoai)ON DELETE CASCADE,
  FOREIGN KEY (MaChucNang) REFERENCES CHUCNANG(MaChucNang),
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV)
);


CREATE TABLE NGUOIDUNG
( MaNguoiDung INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,  
  TenDangNhap nvarchar2 (255) NOT NULL,
    CONSTRAINT TenDN_Unique UNIQUE (TenDangNhap),
  TenHienThi NVARCHAR2(255) NOT NULL ,
  MatKhau nvarchar2 (255) NOT NULL CONSTRAINT constraint_MK CHECK (REGEXP_LIKE(MatKhau, '^[a-zA-Z0-9@$!%*#?&]{8,}$')),
  Quyen nvarchar2 (20) NOT NULL,
  Email nvarchar2 (255) CONSTRAINT constraint_email CHECK (REGEXP_LIKE(Email, '^[A-Za-z]+[A-Za-z0-9.]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,4}$') AND LENGTH(Email) >= 0 AND LENGTH(Email) <= 255),
    CONSTRAINT Email_Unique UNIQUE (Email),
  SoDienThoai nvarchar2 (14) CONSTRAINT constraint_sdt CHECK (REGEXP_LIKE(SoDienThoai, '^(0|84)\d{9,12}$') AND LENGTH(SoDienThoai) >= 10 AND LENGTH(SoDienThoai) <= 14),
    CONSTRAINT sdt_Unique UNIQUE (SoDienThoai),
  HinhAnh nvarchar2 (255)
);


CREATE TABLE YEUTHICH
( YeuThich nvarchar2 (10) NOT NULL,
  ThoiGianYeuThich TIMESTAMP,
  MaNguoiDung INTEGER,
  MaPHCN INTEGER,
  MaChucNang INTEGER,
  MaPhienBan INTEGER,
  MaDV INTEGER,
  MaHDSD INTEGER,
  FOREIGN KEY (MaNguoiDung) REFERENCES NGUOIDUNG(MaNguoiDung)ON DELETE CASCADE,
  FOREIGN KEY (MaPHCN) REFERENCES PHANHECHUCNANG(MaPHCN),
  FOREIGN KEY (MaChucNang) REFERENCES CHUCNANG(MaChucNang),
  FOREIGN KEY (MaPhienBan) REFERENCES PHIENBANHDSD(MaPhienBan),
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV),
  FOREIGN KEY (MaHDSD) REFERENCES HUONGDANSUDUNG(MaHDSD)
);

CREATE TABLE BINHLUAN
( MABL INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,
  NoiDung nvarchar2 (1024) NOT NULL,
  TraLoi integer default null,
  ThoiGianBinhLuan TIMESTAMP NOT NULL,
  ThongBao nvarchar2 (20) default NULL,
  TraLoiMaND integer default NULL,
  MaNguoiDung INTEGER,
  MaPHCN INTEGER,
  MaChucNang INTEGER,
  MaPhienBan INTEGER,
  MaDV INTEGER,
  MaHDSD INTEGER,
  FOREIGN KEY (MaNguoiDung) REFERENCES NGUOIDUNG(MaNguoiDung),
  FOREIGN KEY (MaPHCN) REFERENCES PHANHECHUCNANG(MaPHCN),
  FOREIGN KEY (MaChucNang) REFERENCES CHUCNANG(MaChucNang),
  FOREIGN KEY (MaPhienBan) REFERENCES PHIENBANHDSD(MaPhienBan),
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV),
  FOREIGN KEY (MaHDSD) REFERENCES HUONGDANSUDUNG(MaHDSD)
);


CREATE TABLE CHITIETPHIENBANHDSD
( MoTa nvarchar2 (1024) NOT NULL,
  TenTieuDePB NVARCHAR2(255),
  Revert NVARCHAR2(10) DEFAULT 'true',
  DuongDanFile nvarchar2 (1024) NOT NULL,
  ThoiGianTao TIMESTAMP,
  MaNguoiDung INTEGER,
  MaHDSD INTEGER,
  MaPhienBan INTEGER,
  FOREIGN KEY (MaNguoiDung) REFERENCES NGUOIDUNG(MaNguoiDung),
  FOREIGN KEY (MaPhienBan) REFERENCES PHIENBANHDSD(MaPhienBan),
  FOREIGN KEY (MaHDSD) REFERENCES HUONGDANSUDUNG(MaHDSD)
);



CREATE TABLE PHANQUYEN
( MaNguoiDung INTEGER,
  MaDV INTEGER,
  FOREIGN KEY (MaNguoiDung) REFERENCES NGUOIDUNG(MaNguoiDung),
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV)
);


CREATE TABLE THONGBAO
( MaThongBao INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY NOT NULL PRIMARY KEY,
  ThoiGianThongBao TIMESTAMP NOT NULL,
  NoiDungTB nvarchar2 (1024) NOT NULL,
  TrangThaiTB nvarchar2 (50) NOT NULL,
  MaNguoiDungGui integer default NULL,
  MaNguoiDung INTEGER,
  MaPHCN INTEGER,
  MaChucNang INTEGER,
  MaPhienBan INTEGER,
  MaDV INTEGER,
  MaHDSD INTEGER,
  MABL INTEGER,
  FOREIGN KEY (MABL) REFERENCES BINHLUAN(MABL),
  FOREIGN KEY (MaNguoiDung) REFERENCES NGUOIDUNG(MaNguoiDung),
  FOREIGN KEY (MaPHCN) REFERENCES PHANHECHUCNANG(MaPHCN),
  FOREIGN KEY (MaChucNang) REFERENCES CHUCNANG(MaChucNang),
  FOREIGN KEY (MaPhienBan) REFERENCES PHIENBANHDSD(MaPhienBan),
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV),
  FOREIGN KEY (MaHDSD) REFERENCES HUONGDANSUDUNG(MaHDSD)
);

CREATE TABLE HASHTAG (
  MaPHCN INTEGER,
  MaDV INTEGER,
  MaLoai INTEGER,
  MaChucNang INTEGER,
  MaPhienBan Integer,
  MaHDSD Integer,
  FOREIGN KEY (MaPHCN) REFERENCES PHANHECHUCNANG(MaPHCN)ON DELETE CASCADE,
  FOREIGN KEY (MaDV) REFERENCES DICHVUSO(MaDV)ON DELETE CASCADE,
  FOREIGN KEY (MaLoai) REFERENCES LOAIHDSD(MaLoai)ON DELETE CASCADE,
  FOREIGN KEY (MaChucNang) REFERENCES CHUCNANG(MaChucNang),
  FOREIGN KEY (MaPhienBan) REFERENCES PHIENBANHDSD(MaPhienBan)ON DELETE CASCADE,
  FOREIGN KEY (MaHDSD) REFERENCES HUONGDANSUDUNG(MaHDSD),
  TagName NVARCHAR2(4000)
);
